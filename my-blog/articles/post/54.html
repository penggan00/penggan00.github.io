挂载硬盘

mkdir /mnt/

# 取消挂载的硬盘

umount /dev/sdb1

# 格式化为 ext4

mkfs.ext4 /dev/sdb1

# 查看准备挂载硬盘

lsblk

mkdir /mnt/hp2

mount /dev/sdb1 /mnt/hp2

# 开启自动挂载硬盘

# 获取uuid

blkid /dev/sdb1

nano /etc/fstab

UUID=e0280430-4e37-471a-b784-ad948d7e48f4 /mnt/hp auto nosuid,nodev,nofail,x-gvfs-show 0 0

UUID=5ddaba30-1e6b-448b-b563-5b823c7f2e10 /media/devmon/sdb1-usb-HP_SSD_S_700_500 auto nosuid,nodev,nofail,x-gvfs-show 0 0

chmod -R 755 /mnt/hp2
