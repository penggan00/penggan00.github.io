loli脚本： 

BLOCK:HttpRequest
                url = $"http:///cdn-cgi/trace"
                maxNumberOfRedirects = 2
                customHeaders = {("Host", "cf.nicename.tk")}
                timeoutMilliseconds = 3000
                TYPE:STANDARD
                $""
                "application/x-www-form-urlencoded"
              ENDBLOCK
              
              BLOCK:Keycheck
                banIfNoMatch = False
                KEYCHAIN SUCCESS OR
                  STRINGKEY @data.SOURCE Contains "h=cf.nicename.tk"
              ENDBLOCK
